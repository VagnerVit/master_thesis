% Stroke phases of front crawl — side view stick figures
\begin{tikzpicture}[
    swimmer/.style={thick, line cap=round, line join=round},
    water/.style={blue!20, opacity=0.3},
    arrow/.style={-{Stealth[length=3mm]}, thick, gray!70},
    phase label/.style={font=\bfseries\small, anchor=south},
    propulsive/.style={fill=red!10, draw=red!30, rounded corners=3pt},
]

% Water line
\fill[water] (-0.5, -1.5) rectangle (15.5, 0);
\draw[blue!40, thick] (-0.5, 0) -- (15.5, 0);

% Phase A: Entry & Catch (x=0..3.5)
\begin{scope}[shift={(1.5, 0)}]
    \node[phase label] at (0, 1.8) {A: Entry/Catch};
    % Head
    \draw[swimmer] (0, 0.8) circle (0.25);
    % Torso
    \draw[swimmer] (0, 0.55) -- (-0.3, -0.6);
    % Leading arm (entering water, extended forward)
    \draw[swimmer, blue!70!black] (0, 0.4) -- (1.2, -0.2);
    % Trailing arm (along body)
    \draw[swimmer] (0, 0.4) -- (-0.8, -0.1);
    % Legs
    \draw[swimmer] (-0.3, -0.6) -- (-0.8, -1.2);
    \draw[swimmer] (-0.3, -0.6) -- (-0.6, -1.3);
    % Entry arrow
    \draw[-{Stealth}, blue!70!black, dashed] (0.8, 0.5) -- (1.2, -0.1);
\end{scope}

% Phase B: Pull (x=4..7.5)
\begin{scope}[shift={(5.5, 0)}]
    \draw[propulsive] (-1.2, -1.6) rectangle (1.2, 2.1);
    \node[phase label, red!70!black] at (0, 1.8) {B: Pull};
    % Head
    \draw[swimmer] (0, 0.8) circle (0.25);
    % Torso
    \draw[swimmer] (0, 0.55) -- (-0.2, -0.6);
    % Pulling arm (under body, bent elbow)
    \draw[swimmer, red!70!black, very thick] (0, 0.4) -- (0.3, -0.1) -- (-0.1, -0.5);
    % Other arm (recovery, above water)
    \draw[swimmer] (0, 0.4) -- (-0.6, 0.9);
    % Legs
    \draw[swimmer] (-0.2, -0.6) -- (-0.6, -1.2);
    \draw[swimmer] (-0.2, -0.6) -- (-0.4, -1.3);
    % Force arrow
    \draw[-{Stealth}, red!70!black, thick] (-0.1, -0.5) -- (-0.1, -1.0) node[right, font=\scriptsize] {$F$};
\end{scope}

% Phase C: Push (x=8..11.5)
\begin{scope}[shift={(9.5, 0)}]
    \draw[propulsive] (-1.2, -1.6) rectangle (1.2, 2.1);
    \node[phase label, red!70!black] at (0, 1.8) {C: Push};
    % Head
    \draw[swimmer] (0, 0.8) circle (0.25);
    % Torso
    \draw[swimmer] (0, 0.55) -- (-0.2, -0.6);
    % Pushing arm (extending back)
    \draw[swimmer, red!70!black, very thick] (0, 0.4) -- (-0.2, -0.2) -- (-0.8, -0.6);
    % Other arm (forward, entering)
    \draw[swimmer] (0, 0.4) -- (0.7, 0.6);
    % Legs
    \draw[swimmer] (-0.2, -0.6) -- (-0.7, -1.1);
    \draw[swimmer] (-0.2, -0.6) -- (-0.3, -1.3);
    % Force arrow
    \draw[-{Stealth}, red!70!black, thick] (-0.8, -0.6) -- (-1.3, -0.6) node[left, font=\scriptsize] {$F$};
\end{scope}

% Phase D: Recovery (x=12..15.5)
\begin{scope}[shift={(13.5, 0)}]
    \node[phase label] at (0, 1.8) {D: Recovery};
    % Head
    \draw[swimmer] (0, 0.8) circle (0.25);
    % Torso
    \draw[swimmer] (0, 0.55) -- (-0.3, -0.6);
    % Recovery arm (high elbow above water)
    \draw[swimmer, blue!70!black] (0, 0.4) -- (0.2, 1.1) -- (0.8, 0.7);
    % Other arm (extended forward underwater)
    \draw[swimmer] (0, 0.4) -- (1.0, -0.1);
    % Legs
    \draw[swimmer] (-0.3, -0.6) -- (-0.8, -1.1);
    \draw[swimmer] (-0.3, -0.6) -- (-0.5, -1.3);
    % Recovery arc
    \draw[-{Stealth}, blue!70!black, dashed] (0.8, 0.7) arc (30:150:0.5);
\end{scope}

% Arrows between phases
\draw[arrow] (3.0, 0.5) -- (4.0, 0.5);
\draw[arrow] (7.0, 0.5) -- (8.0, 0.5);
\draw[arrow] (11.0, 0.5) -- (12.0, 0.5);

% Direction of swimming
\draw[-{Stealth}, thick, black!60] (0, -1.8) -- (2.5, -1.8) node[right, font=\small] {direction of swimming};

\end{tikzpicture}
